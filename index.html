Wiki Pages

A wiki page is a simple text file composed for front-matter and markdown. The front-matter specifies meta information and selects the theme wikismith will use to render the page.

Gulpfile

var app = express();
var lr = tinylr();

function serve() {
    wikismith.watch()
        .pipe(wikismith.pipeline())
        .pipe(gulp.dest('build'))
        .pipe(es.map(function(file, cb) {
            lr.changed({body: { files: [file.path] }});
            cb(null, file);
        }));

    app.use(livereload());
    app.use(express.static(__dirname + '/build'));
    app.listen(9292);
    lr.listen(35729);
}

gulp.task('default', function() {
    serve()
});
